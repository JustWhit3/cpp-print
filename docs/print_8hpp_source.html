640  template <typename T>
641  static constexpr bool is_escape( const T& str, ANSI&& flag )
642  {
643  if constexpr( std::is_convertible_v <T, stype::string_view<T_str>> && ! std::is_same_v<T, std::nullptr_t> )
644  {
645  switch( flag )
646  {
647  case( ANSI::first ):

680  // print_backend
691  template <class T_os, class T, class... Args>
692  void print_backend( T_os&& os, T&& first, Args&&... args ) const
693  {
694  std::lock_guard <std::mutex> lock{ mutex_ };
695 
696  // Printing the first argument
697  if( is_null_str( getPattern() ) || is_null_str( first ) || is_escape( first, ANSI::first ) ) os << first;
698  else os << getPattern() << first << getPattern();
699 
700  // Printing all the other arguments
701  if constexpr( sizeof...( args ) > 0 )
702  {
703  if ( is_null_str( first ) || is_escape( first, ANSI::first ) )
704  {
705  if( is_null_str( getPattern() ) ) ( ( os << args << getSep() ), ...);
706  else ( ( os << getPattern() << args << getPattern() << getSep() ), ...);
707  }
708  else
709  {
710  if( is_null_str( getPattern() ) ) ( ( os << getSep() << args ), ...);
711  else ( ( os << getSep() << getPattern() << args << getPattern() ), ...);
712  }
713  }
714  os << getEnd();
715 
716  // Resetting the stream from ANSI escape sequences
717  if constexpr( sizeof...( args ) > 0 )
718  {
719  if ( is_escape( first, ANSI::generic ) || ( ( is_escape( args, ANSI::generic ) ) || ...) )
720  {
721  os << reset_ANSI;
722  }
723  }
724  else
725  {
726  if ( is_escape( first, ANSI::generic ) )
727  {
728  os << reset_ANSI;
729  }
730  }
731

733  if ( getFlush() && ! std::is_base_of_v <stype::ostringstream<T_str>, T_os> ) os << std::flush;
734  }
