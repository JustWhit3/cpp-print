598  template <class T_os, class T, class... Args>
599  void print_backend( T_os&& os, T&& first, Args&&... args ) const
600  {
601  std::lock_guard <std::mutex> lock{ mutex_ };
602 
603  // Printing the first argument
604  if( is_null_str( getPattern() ) || is_null_str( first ) || is_escape( first, ANSI::first ) ) os << first;
605  else os << getPattern() << first << getPattern();
606 
607  // Printing all the other arguments
608  if constexpr( sizeof...( args ) > 0 )
609  {
