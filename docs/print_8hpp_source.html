548  // print_backend
559  template <class T_os, class T, class... Args>
560  void print_backend( T_os&& os, T&& first, Args&&... args ) const
561  {
562  std::lock_guard <std::mutex> lock{ mutex_ };
563 
564  // Printing the first argument
565  if( is_null_str( getPattern() ) || is_null_str( first ) || is_escape( first, ANSI::first ) ) os << first;
566  else os << getPattern() << first << getPattern();
567 
568  // Printing all the other arguments
569  if constexpr( sizeof...( args ) > 0 )
570  {
571  if ( is_null_str( first ) || is_escape( first, ANSI::first ) )
572  {
573  if( is_null_str( getPattern() ) ) ( ( os << args << getSep() ), ...);
574  else ( ( os << getPattern() << args << getPattern() << getSep() ), ...);
575  }
576  else
