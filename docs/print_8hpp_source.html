651  template <typename T>
652  static constexpr bool is_escape( const T& str, ANSI&& flag )
653  {
654  if constexpr( std::is_convertible_v <T, stype::string_view<T_str>> && ! std::is_same_v<T, std::nullptr_t> )
655  {
656  switch( flag )
657  {

691  // print_backend
702  template <class T_os, class T, class... Args>
703  void print_backend( T_os&& os, T&& first, Args&&... args ) const
704  {
705  std::lock_guard <std::mutex> lock{ mutex_ };
706 
707  // Printing the first argument
708  if( is_null_str( getPattern() ) || is_null_str( first ) || is_escape( first, ANSI::first ) ) os << first;
709  else os << getPattern() << first << getPattern();
710 
711  // Printing all the other arguments
712  if constexpr( sizeof...( args ) > 0 )
713  {
714  if ( is_null_str( first ) || is_escape( first, ANSI::first ) )
715  {
716  if( is_null_str( getPattern() ) ) ( ( os << args << getSep() ), ...);
717  else ( ( os << getPattern() << args << getPattern() << getSep() ), ...);
718  }
719  else
720  {
721  if( is_null_str( getPattern() ) ) ( ( os << getSep() << args ), ...);
722  else ( ( os << getSep() << getPattern() << args << getPattern() ), ...);
723  }
724  }
725  os << getEnd();
726 
727  // Resetting the stream from ANSI escape sequences
728  if constexpr( sizeof...( args ) > 0 )
729  {
730  if ( is_escape( first, ANSI::generic ) || ( ( is_escape( args, ANSI::generic ) ) || ...) )
731  {
732  os << reset_ANSI;
733  }
734  }
735  else
736  {
737  if ( is_escape( first, ANSI::generic ) )
738  {
739  os << reset_ANSI;
740  }
741  }

744  if ( getFlush() && ! std::is_base_of_v <stype::ostringstream<T_str>, T_os> ) os << std::flush;
745  }
