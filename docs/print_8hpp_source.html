408  // print_backend

420  void print_backend( T_os&& os, T&& first, Args&&... args ) const
421  {
422  std::lock_guard <std::mutex> lock{ mutex_ };
423 
424  // Printing all the arguments

428  if ( is_null_str( first ) || is_escape( first, ANSI::first ) ) ( ( os << args << getSep() ), ...);
