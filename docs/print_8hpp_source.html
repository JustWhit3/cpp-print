665  // print_backend
676  template <class T_os, class T, class... Args>
677  void print_backend( T_os&& os, T&& first, Args&&... args ) const
678  {
679  std::lock_guard <std::mutex> lock{ mutex_ };
680 
681  // Printing the first argument
682  if( is_null_str( getPattern() ) || is_null_str( first ) || is_escape( first, ANSI::first ) ) os << first;
683  else os << getPattern() << first << getPattern();
684 
685  // Printing all the other arguments
686  if constexpr( sizeof...( args ) > 0 )
687  {
688  if ( is_null_str( first ) || is_escape( first, ANSI::first ) )
689  {
690  if( is_null_str( getPattern() ) ) ( ( os << args << getSep() ), ...);
691  else ( ( os << getPattern() << args << getPattern() << getSep() ), ...);
692  }
693  else

695  if( is_null_str( getPattern() ) ) ( ( os << getSep() << args ), ...);
696  else ( ( os << getSep() << getPattern() << args << getPattern() ), ...);
697  }

703  if constexpr( sizeof...( args ) > 0 )
704  {
705  if ( is_escape( first, ANSI::generic ) || ( ( is_escape( args, ANSI::generic ) ) || ...) )
